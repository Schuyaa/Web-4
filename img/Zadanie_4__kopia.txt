#include <iostream>
#include <fstream>
#include <stdlib.h>

using namespace std;

int create(int* A, int n)
{
	ifstream f;
	f.open("test.txt");
	for (int i = 0; i < n; i++) f >> A[i];
	return *A;
	f.close();
}
void enter(int* A, int n) {
	for (int i = 0; i < n; i++)
	{
		cout << A[i] << ' ';
	}
}
int counter(int* A, int n)
{
	int cnt;
	cnt = 0;
	for (int i = 0; i < n; i++)
	{
		if (A[i] % 2 == 0)
		{
			cnt += 1;
		}
	}
	return cnt;
}
int minimum(int* A, int n)
{
	int min_nechet;
	min_nechet = A[0];
	for (int i = 0; i < n; i++)
	{
		if (min_nechet > A[i] && A[i] % 2 == 1)
		{
			min_nechet = A[i];
		}
	}
	return min_nechet;

}
void del_chet(int*&A, int &n, int cnt_chet)
{
	n = n - cnt_chet;
	int j = 0;
	int* B = new int[n];
	for (int i = 0; i < n+cnt_chet; i++) if (A[i] % 2 == 1)
		{
			B[j] = A[i];
			j++;
		}
	delete[] A;

	A = B;
}
int add(int*&A, int &n, int m)
{
	n++;
	int* B = new int[n];
	for (int i = 0; i < n-1; i++)
	{
		B[i] = A[i];
		if (A[i] == m) 
		{
			B[i+1] = -1;
			for (int j = i+1; j < n; j++)
			{
				B[j+1] = A[j];


			}
			delete[] A;
			A = B;
			return 0;
		}
	}
}

int main()
{
	setlocale(LC_ALL, "ru");
	int n, m, cnt_chet;
	cout << "Введите размер массива: ";
	cin >> n;
	int* A = new int[n];

	create(A, n);
	cout << "Массив: ";
	enter(A, n);
	cout << "\n";

	m = minimum(A, n);
	cnt_chet = counter(A, n);

	del_chet(A, n, cnt_chet);
	cout << "Массив после удаления: ";
	enter(A, n);
	cout << "\n";

	add(A, n, m);
	cout << "Массив после добавления: ";
	enter(A, n);


}
